function searchFood(){

let input = document.getElementById("foodInput").value.toLowerCase().trim();

let food = foods[input];

let result = document.getElementById("result");

result.style.display = "block";

if(food){

document.getElementById("foodName").innerText = input.toUpperCase();
document.getElementById("calories").innerText = food.calories;
document.getElementById("protein").innerText = food.protein;
document.getElementById("carbs").innerText = food.carbs;
document.getElementById("sugar").innerText = food.sugar;
document.getElementById("fiber").innerText = food.fiber;
document.getElementById("spike").innerText = food.spike;
document.getElementById("score").innerText = food.score + "/100";
document.getElementById("tip").innerText = food.tip;

}else{

document.getElementById("foodName").innerText = "Food not found";
document.getElementById("calories").innerText = "-";
document.getElementById("protein").innerText = "-";
document.getElementById("carbs").innerText = "-";
document.getElementById("sugar").innerText = "-";
document.getElementById("fiber").innerText = "-";
document.getElementById("spike").innerText = "-";
document.getElementById("score").innerText = "-";
document.getElementById("tip").innerText = "Try apple, banana, rice, pizza";
}

}
